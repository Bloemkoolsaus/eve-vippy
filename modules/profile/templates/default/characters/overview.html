Authorized Characters

{assign var=i value=0} {foreach from=$characters key=c item=character} {if $character->isAuthorized()}
{include file="profile/characters/character" character=$character}

Un-Authorized Characters

{assign var=i value=0} {foreach from=$characters key=c item=character} {if !$character->isAuthorized()}
{include file="profile/characters/character" character=$character}
{assign var=i value=$i+1} {/if} {/foreach}
